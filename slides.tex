\begin{frame}{Learning With Errors}
\includegraphics[scale=0.47]{images/LWE.jpg}

\pause
\bigskip
Есть фиксированный вектор $\vec s \in \Z_q^n$.

\pause
\bigskip
Составляется система зашумлённых линейных уравнений:
\begin{equation*}
	A\vec s+\vec e = \vec b \pmod q
\end{equation*}

\begin{itemize}
	\item $A$ --- матрица из $\Z_q^{m \times n}$
	\item $\vec e$ --- вектор из $\Z_q^m$
\end{itemize}

\pause
\bigskip
По известным $(A, \vec b)$ нужно восстановить $\vec s$.
\end{frame}

\begin{frame}{LWE-problem}
Зафиксируем некоторое распределение $\chi$.

\pause
\bigskip
Будем генерировать:
\begin{itemize}
	\item $\vec s$ --- из $\chi^n$
	\item $A$ --- равномерно из $\Z_q^{m \times n}$
	\item $\vec e$ --- из $\chi^m$
\end{itemize}

\pause
\bigskip
Установим максимально допустимую вероятность ошибки $\varepsilon$.

\pause
\bigskip
Задача --- по данным $(A, \vec b)$ с $\Pr \geq 1-\varepsilon$ находить $\vec s$.
\end{frame}

\begin{frame}{RLWE-problem}
	Обозначим $\mathcal{R}_q^n = \Z_q[x]/(x^n+1)$.
	
	\medskip
	Вместо $A, \vec s, \vec e$ будут $\bold a, \bold s, \bold e \in \mathcal{R}_q^n$.
	
	\pause
	\begin{equation*}
		\bold a\bold s+\bold e = \bold b
	\end{equation*}
	
	\begin{itemize}
		\item $\bold s, \bold e$ --- генерируются из $\chi^n$.
		\item $\bold a$ --- равномерно из $\mathcal{R}_q^n$.
	\end{itemize}
	
	\pause
	\bigskip	
	По известным $(\bold a, \bold b)$ нужно c $\Pr \geq 1-\varepsilon$ восстанавливать $\bold s$.
\end{frame}

\begin{frame}{RLWE-шифрование}
\begin{minipage}{0.47\textwidth}
	\uncover<1->{
	Выбираем распределение $\chi_k$.}
	
	\begin{itemize}
		\item Открытый ключ $(\bold a, \bold b)$
		\item Закрытый ключ $\bold s$
	\end{itemize}

	\bigskip	
	\uncover<2->{
	Знаем открытый ключ, получаем\\ шифровку сообщения $(\bold u, \bold v)$.}
	
	\bigskip\smallskip
	\uncover<3->{
	Знаем закрытый ключ, можем\\
	восстановить само сообщение.
	
	\smallskip
	Взлом из решения $\mathsf{RLWE}$-problem.}
\end{minipage}
\hfill
\begin{minipage}{0.41\textwidth}
	\uncover<1->{
	Генерация ключей
	\begin{algorithmic}[1]
		\item $\bold a \leftarrow \mathcal{R}_q^n$
		\item $\bold s, \bold e \leftarrow \chi_k^n$
		\item $\bold b = \bold a\bold s+\bold e$
	\end{algorithmic}
	}
	
	\medskip
	\uncover<2->{
	Зашифрование
	\begin{algorithmic}[1]
		\item $\bold{s'}, \bold{e'}, \bold{e''} \leftarrow \chi_k^n$
		\item $\bold u = \bold a\bold{s'}+\bold{e'}$
		\item $\bold v = \bold b\bold{s'}+\bold{e''}+\Map(m)$
	\end{algorithmic}
	}
	
	\medskip
	\uncover<3->{
	Расшифрование
	\begin{algorithmic}[1]
		\item $m' = \Demap(\bold v-\bold u\bold s)$
	\end{algorithmic}
	}
\end{minipage}

\only<4->{
\begin{align*}
	\bold v-\bold u\bold s = \Map(m)+(\bold b\bold{s'}+\bold{e''})-(\bold a\bold{s'}+\bold{e'})\bold s & =\\[-13pt]
	= \Map(m)+(\bold a\bold s+\bold e)\bold{s'}+\bold{e''}-\bold a\bold{s'}\bold s-\bold{e'}\bold s & = \Map(m)+\overset{\bold z}{\overbrace{\bold e\bold{s'}-\bold{e'}\bold s+\bold{e''}}}
\end{align*}
}
\end{frame}
